### 保持canvas绘制的图片居中
```
wxx.image.info("https://qschou-cdn.oss-cn-hangzhou.aliyuncs.com/images/wxapp/find/img1.jpg").then(res => {
    /* 获取 canvas 上下文 */
    const ctx = wx.createCanvasContext("mycanvas");
    /* 路径/宽/高 */
    let {width, height, path} = res;
    /* 定义显示的宽高比 */
    let ratio = 1.25;  /** 示例宽高比为1.25 */
    /* 宽高比不同, 不同方式处理 */
    if(width/height > ratio){
        ctx.drawImage(path, (width-ratio*height)/2, 0, height*ratio, height, 0, 0, 300, 240);
    }else{
        ctx.drawImage(path, 0, (height-width/ratio)/2, width, width/ratio, 0, 0, 300, 240);
    }
    /* 画 */
    ctx.draw();
});
```
